1.绝对定位

.position_relative{
    position:relative; 
}
.position_absolute{
    position: absolute;
    margin:auto;
    top: 0;
    left: 0;
    right: 0;
    bottom: 0;
}
<div class="out_box position_relative">
    <div class="in_box position_absolute"></div>
</div>

2.display:table-cell

}

    vertical-align: middle;
}

    </div>
</div>

</div>
